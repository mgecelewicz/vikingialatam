        function createCinematicSystem(texture) {

            const aspect = texture.image ? texture.image.width / texture.image.height : 1;

            const sharedUniforms = {

                uProgress: { value: 0 },

                uViking: { value: 0 },

                uStrike: { value: 0 },

                uTime: { value: 0 },

                uAlpha: { value: 1.0 }

            };

            // Rostro

            const faceMat = new THREE.ShaderMaterial({

                uniforms: { ...sharedUniforms, uTexture: { value: texture }, uAlpha: { value: 1.0 } },

                vertexShader, fragmentShader, transparent: true, side: THREE.DoubleSide

            });

            faceParticles = new THREE.Mesh(createParticleGeometry(15 * aspect, 15, 180, true), faceMat);

            scene.add(faceParticles);

            // Texto

            const textTexture = createTextTexture();

            const textMat = new THREE.ShaderMaterial({

                uniforms: { ...sharedUniforms, uTexture: { value: textTexture }, uAlpha: { value: 0.0 } },

                vertexShader, fragmentShader, transparent: true, side: THREE.DoubleSide

            });

            textParticles = new THREE.Mesh(createParticleGeometry(22, 11, 140, false), textMat);

            textParticles.position.z = 2.0;

            scene.add(textParticles);

            setupCinematicTimeline(faceMat, textMat);

        }

        function createParticleGeometry(w, h, res, isViking) {

            const geo = new THREE.PlaneGeometry(w, h, res, res).toNonIndexed();

            const count = geo.attributes.position.count;

            const randoms = new Float32Array(count);

            const centers = new Float32Array(count * 3);

            const targets = new Float32Array(count * 3);

            for(let i = 0; i < count; i += 3) {

                const r = Math.random();

                const cx = geo.attributes.position.getX(i);

                const cy = geo.attributes.position.getY(i);

               

                let tx, ty, tz = -4.0;

               

                if(isViking) {

                    if (r > 0.85) {

                        tx = (Math.random() - 0.5) * 8.0;

                        ty = 4.0 + Math.random() * 4.0;

                    } else if (r > 0.70) {

                        const side = Math.random() > 0.5 ? 1 : -1;

                        tx = side * (3.0 + Math.random() * 5.0);

                        ty = 7.0 + Math.random() * 4.0;

                    } else {

                        tx = (Math.random() - 0.5) * 10.0;

                        ty = -2.0 + (Math.random() - 0.5) * 12.0;

                    }

                } else {

                    tx = cx; ty = cy;

                }

                for(let j = 0; j < 3; j++) {

                    randoms[i+j] = r;

                    centers[(i+j)*3] = cx;

                    centers[(i+j)*3+1] = cy;

                    centers[(i+j)*3+2] = (Math.random()-0.5) * 5.0;

                    targets[(i+j)*3] = tx;

                    targets[(i+j)*3+1] = ty;

                    targets[(i+j)*3+2] = tz;

                }

            }

            geo.setAttribute('aRandom', new THREE.BufferAttribute(randoms, 1));

            geo.setAttribute('aCenter', new THREE.BufferAttribute(centers, 3));

            geo.setAttribute('aTarget', new THREE.BufferAttribute(targets, 3));

            return geo;

        }
